## Instalation

### npm

```sh
npm i react-css-spinners
```

### yarn

```sh
yarn add react-css-spinners
```

### unpkg

```html
<script src="https://unpkg.com/react-ui-spinners@latest/dist/bundle.umd.js"></script>
```

```js

import { Ellipsis  } from 'react-ui-spinners';
```
